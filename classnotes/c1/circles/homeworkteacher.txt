\begin{enumerate}
    \newpage %Question 1
    \prob
    Find the coordinates of the radius and the centre of the circle with equation:
        \begin{enumerate}
        \prob[2] $(x-3)^2 + (y+4)^2 =25$
        \prob[2] $x^2 + (y-4)^2 = 16$
        \prob[3] $x^2 + y^2 - 4x + 6y + 9 = 0$
        \end{enumerate}
    % Question 2
    \prob
    Write down the equation of the circle with the following features:
        \begin{enumerate}
        \prob[2] centre $= (0,0),$ radius $= 5$
        \prob[2] centre $= (2,3),$ radius $=\sqrt{5}$
        \prob[2] centre $=(-4,3),$ radius $=6.$
        \end{enumerate}
    
    \newpage %Question 3
    \prob
    The circle $C$ is given by the equation $x^2 + y^2 -6x + 8y+9=0.$
        \begin{enumerate}
        \prob[2] Find the coordinates of the centre and the radius of $C.$
        \prob[3] Find the exact coordinates of the points where the $C$ crosses both axes.
        \prob[1] Find, in terms of $\pi,$ the area and circumference of $C.$
        \end{enumerate}
    
    \newpage %Question 4
    \prob[3]
    Describe the geometrical transformation which has been applied to the circle with equation $x^2 + y^2 = 25$ to obtain the circle with equation $x^2 + y^2 -2x +4y -20 = 0.$

    \newpage %Question 5
    \prob[4]
    $A$ and $B$ are the coordinates of the end points of the diameter $AB$ of a circle, where $A = (7,-3)$ and $B = (1,-11).$ Find the equation of the circle.
    
    \newpage %Question 6
    \prob
    A triangle $PQR$ has vertices $P(-1,5), Q(7,1)$ and $R(-5,-3).$
        \begin{enumerate}
        \prob[2]
        Prove that triangle $PQR$ is right-angled.
        \prob[4] Find the equation of the circle which passes through the points $P,Q$ and $R.$
        \end{enumerate}
        
    \newpage %Question 7
    \prob
    The circle $C$ is given by the equation $x^2 + y^2 -4x + 10y + 4 = 0.$
        \begin{enumerate}
        \prob[2] Find the coordinates of the centre and the radius of the circle.
        \prob[4] Hence sketch the circle.
        \prob[1] Show that $P(6,-2)$ lies on the circle.
        \prob[3] Find the equation of the tangent to the circle at the point $P,$ giving your answer in the form $ax+by+c=0$ where $a,b,c$ are integers.
        \end{enumerate}
    
    \newpage %Question 8
    \prob
        The equation of a circle is given by $x^2 + y^2 - 4x + 2y + 2 = 0.$
        \begin{enumerate}
        \prob[2] Find the coordinates of the centre and the radius of the circle.
        \prob[3]
        $P$ is a point outside the circle whose coordinates are $(5,-5).$ Find the exact value of the lengths of the tangents drawn from $P$ to the circle.
        \end{enumerate}
    
\end{enumerate}
